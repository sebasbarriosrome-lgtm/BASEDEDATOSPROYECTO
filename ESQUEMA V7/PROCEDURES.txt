-- =====================================================
-- SP: REGISTRAR USUARIO
-- =====================================================
CREATE PROCEDURE sp_register_usuario(
    IN p_nombre VARCHAR(20),
    IN p_apellido VARCHAR(20),
    IN p_email VARCHAR(35),
    IN p_password VARCHAR(300)
)
BEGIN

    DECLARE v_count INT;

    -- VALIDAR EMAIL DUPLICADO
    SELECT COUNT(*)
    INTO v_count
    FROM usuario
    WHERE email = p_email;

    IF v_count > 0 THEN
        SIGNAL SQLSTATE '45000'
        SET MESSAGE_TEXT = 'El correo ya está registrado';
    END IF;

    -- INSERTAR USUARIO
    INSERT INTO usuario (
        nombre,
        apellido,
        email,
        password
    )
    VALUES (
        p_nombre,
        p_apellido,
        p_email,
        p_password
    );

END $$


-- =====================================================
-- SP: OBTENER USUARIO POR EMAIL
-- =====================================================
CREATE PROCEDURE sp_get_usuario_by_email(
    IN p_email VARCHAR(35)
)
BEGIN

    SELECT 
        id,
        nombre,
        apellido,
        email,
        password,
        rol_global,
        created_at,
        updated_at,
        deleted_at
    FROM usuario
    WHERE email = p_email
    AND deleted_at IS NULL
    LIMIT 1;

END $$


-- =====================================================
-- SP: ELIMINAR USUARIO (SOFT DELETE)
-- =====================================================
CREATE PROCEDURE sp_eliminar_usuario(
    IN p_id BIGINT
)
BEGIN

    DECLARE v_count INT;

    -- VALIDAR QUE EXISTA
    SELECT COUNT(*)
    INTO v_count
    FROM usuario
    WHERE id = p_id
    AND deleted_at IS NULL;

    IF v_count = 0 THEN
        SIGNAL SQLSTATE '45000'
        SET MESSAGE_TEXT = 'El usuario no existe o ya fue eliminado';
    END IF;

    -- SOFT DELETE
    UPDATE usuario
    SET deleted_at = CURRENT_TIMESTAMP
    WHERE id = p_id;

END $$

CREATE PROCEDURE sp_create_club(
    IN p_nombre VARCHAR(50),
    IN p_ciudad VARCHAR(20),
    IN p_descripcion TEXT,
    IN p_logo_url TEXT,          -- ✅ CAMBIADO
    IN p_banner_url TEXT,        -- ✅ CAMBIADO
    IN p_color_primario VARCHAR(20),
    IN p_color_secundario VARCHAR(20),
    IN p_contacto VARCHAR(20)
)
BEGIN

    DECLARE v_count INT;
    DECLARE v_slug VARCHAR(50);

    -- ✅ Validar duplicado (nombre + ciudad)
    SELECT COUNT(*)
    INTO v_count
    FROM club
    WHERE nombre = p_nombre 
      AND ciudad = p_ciudad
      AND deleted_at IS NULL;

    IF v_count > 0 THEN
        SIGNAL SQLSTATE '45000'
        SET MESSAGE_TEXT = 'El club ya existe en esa ciudad';
    END IF;

    -- ✅ Generar slug (mejorado)
    SET v_slug = LOWER(REPLACE(TRIM(p_nombre), ' ', '-'));

    -- ✅ Insertar club
    INSERT INTO club (
        nombre,
        slug,
        ciudad,
        descripcion,
        logo_url,
        banner_url,
        color_primario,
        color_secundario,
        contacto,
        estado
    )
    VALUES (
        p_nombre,
        v_slug,
        p_ciudad,
        p_descripcion,
        p_logo_url,
        p_banner_url,
        p_color_primario,
        p_color_secundario,
        p_contacto,
        'activo'
    );

END $$

CREATE PROCEDURE sp_update_usuario(
    IN p_email_actual VARCHAR(35),
    IN p_nombre VARCHAR(20),
    IN p_apellido VARCHAR(20),
    IN p_email VARCHAR(35),
    IN p_telefono VARCHAR(15),
    IN p_birthDate DATE,
    IN p_photoUrl VARCHAR(255)
)
BEGIN

    DECLARE v_usuario_id BIGINT;
    DECLARE v_count_email INT;
    DECLARE v_perfil_count INT;

    -- BUSCAR USUARIO
    SELECT id
    INTO v_usuario_id
    FROM usuario
    WHERE email = p_email_actual
    LIMIT 1;

    IF v_usuario_id IS NULL THEN
        SIGNAL SQLSTATE '45000'
        SET MESSAGE_TEXT = 'Usuario no encontrado';
    END IF;

    -- VALIDAR EMAIL DUPLICADO
    SELECT COUNT(*)
    INTO v_count_email
    FROM usuario
    WHERE email = p_email
    AND id != v_usuario_id;

    IF v_count_email > 0 THEN
        SIGNAL SQLSTATE '45000'
        SET MESSAGE_TEXT = 'El correo ya está registrado';
    END IF;

    -- ACTUALIZAR USUARIO
    UPDATE usuario
    SET
        nombre = p_nombre,
        apellido = p_apellido,
        email = p_email
    WHERE id = v_usuario_id;

    -- VERIFICAR SI PERFIL EXISTE
    SELECT COUNT(*)
    INTO v_perfil_count
    FROM perfil_usuario
    WHERE usuario_id = v_usuario_id;

    -- SI NO EXISTE → INSERT
    IF v_perfil_count = 0 THEN

        INSERT INTO perfil_usuario(
    		usuario_id,
		telefono,
		fecha_nacimiento,
    		foto_url,
    		created_at
	)
	VALUES(
	    v_usuario_id,
	    p_telefono,
	    p_birthDate,
	    p_photoUrl,
	    (
        	SELECT created_at
	        FROM usuario
        	WHERE id = v_usuario_id
	    )
	);

    ELSE

        -- ACTUALIZAR PERFIL
        UPDATE perfil_usuario
        SET
            telefono = p_telefono,
            fecha_nacimiento = p_birthDate,
            foto_url = p_photoUrl
        WHERE usuario_id = v_usuario_id;

    END IF;

END $$

CREATE PROCEDURE sp_create_usuario_club(
    IN p_usuario_id BIGINT,
    IN p_club_id BIGINT,
    IN p_rol VARCHAR(20)
)
BEGIN

    DECLARE v_count INT;

    SELECT COUNT(*)
    INTO v_count
    FROM usuario_club
    WHERE usuario_id = p_usuario_id
      AND club_id = p_club_id;

    IF v_count > 0 THEN
        SIGNAL SQLSTATE '45000'
        SET MESSAGE_TEXT = 'El usuario ya pertenece al club';
    END IF;

    INSERT INTO usuario_club(
        usuario_id,
        club_id,
        rol,
        estado,
        created_at
    )
    VALUES(
        p_usuario_id,
        p_club_id,
        p_rol,
        'activo',
        CURRENT_TIMESTAMP
    );

END $$

CREATE PROCEDURE sp_get_panel_club(
    IN p_usuario_id BIGINT
)
BEGIN

    SELECT
    c.id,
    c.nombre,
    c.logo_url,
    c.banner_url,
    c.descripcion,
    c.color_primario,
    c.color_secundario,
    u.nombre,
    p.foto_url
FROM usuario_club uc
JOIN club c ON uc.club_id = c.id
JOIN usuario u ON u.id = uc.usuario_id
LEFT JOIN perfil_usuario p ON p.usuario_id = u.id

WHERE uc.usuario_id = p_usuario_id
AND uc.rol = 'admin'

    LIMIT 1;

END $$

CREATE PROCEDURE sp_create_solicitud(
    IN p_usuario_id BIGINT,
    IN p_club_id BIGINT,
    IN p_rol VARCHAR(20),
    IN p_mensaje TEXT
)
BEGIN

    DECLARE v_exists INT;

    -- ✅ YA PERTENECE A UN CLUB
    SELECT COUNT(*)
    INTO v_exists
    FROM usuario_club
    WHERE usuario_id = p_usuario_id
      AND estado = 'activo';

    IF v_exists > 0 THEN
        SIGNAL SQLSTATE '45000'
        SET MESSAGE_TEXT = 'El usuario ya pertenece a un club';
    END IF;

    -- ✅ YA TIENE SOLICITUD PENDIENTE
    SELECT COUNT(*)
    INTO v_exists
    FROM solicitud
    WHERE usuario_id = p_usuario_id
      AND estado = 'pendiente';

    IF v_exists > 0 THEN
        SIGNAL SQLSTATE '45000'
        SET MESSAGE_TEXT = 'Ya existe una solicitud pendiente';
    END IF;

    INSERT INTO solicitud(
        usuario_id,
        club_id,
        rol_solicitado,
        mensaje,
        estado,
        created_at
    )
    VALUES(
        p_usuario_id,
        p_club_id,
        p_rol,
        p_mensaje,
        'pendiente',
        CURRENT_TIMESTAMP
    );

    SELECT LAST_INSERT_ID() AS solicitud_id;

END $$

CREATE PROCEDURE sp_get_solicitudes_by_club(
    IN p_club_id BIGINT
)
BEGIN

    SELECT
    s.id,
    s.rol_solicitado,
    s.estado,
    s.mensaje,
    u.nombre,

    TIMESTAMPDIFF(YEAR, pu.fecha_nacimiento, CURDATE()) AS edad,

    sd.peso,
    sd.estatura,
    sd.experiencia,
    sd.especialidad

FROM solicitud s
JOIN usuario u ON s.usuario_id = u.id
LEFT JOIN perfil_usuario pu ON pu.usuario_id = u.id
LEFT JOIN solicitud_deportiva sd ON sd.solicitud_id = s.id

WHERE s.club_id = p_club_id
AND s.estado = 'pendiente';


END $$

CREATE PROCEDURE sp_get_horarios_by_club(
    IN p_club_id BIGINT
)
BEGIN

    SELECT
        id,
        dia,
        DATE_FORMAT(hora_inicio, '%H:%i') AS hora_inicio,
