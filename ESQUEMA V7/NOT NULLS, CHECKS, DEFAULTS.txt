ALTER TABLE usuario
MODIFY nombre VARCHAR(20) NOT NULL,
MODIFY apellido VARCHAR(20) NOT NULL,
MODIFY email VARCHAR(35) NOT NULL,
MODIFY password VARCHAR(20) NOT NULL,
MODIFY rol_global ENUM('usuario','admin_plataforma') NOT NULL DEFAULT 'usuario',
MODIFY created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
MODIFY updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP;

ALTER TABLE perfil_usuario
MODIFY usuario_id BIGINT NOT NULL;

ALTER TABLE club
MODIFY nombre VARCHAR(50) NOT NULL,
MODIFY slug VARCHAR(50) NOT NULL,
MODIFY ciudad VARCHAR(20) NOT NULL,
MODIFY estado ENUM('activo','inactivo','suspendido') NOT NULL DEFAULT 'activo',
MODIFY created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP;

ALTER TABLE usuario_club
MODIFY usuario_id BIGINT NOT NULL,
MODIFY club_id BIGINT NOT NULL,
MODIFY rol ENUM('admin','deportista','entrenador') NOT NULL,
MODIFY estado ENUM('activo','retirado','expulsado','pendiente') NOT NULL DEFAULT 'pendiente',
MODIFY created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP;

ALTER TABLE solicitud
MODIFY usuario_id BIGINT NOT NULL,
MODIFY club_id BIGINT NOT NULL,
MODIFY rol_solicitado VARCHAR(20) NOT NULL,
MODIFY estado VARCHAR(10) NOT NULL,
MODIFY created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
MODIFY updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP;

ALTER TABLE solicitud_deportiva
MODIFY solicitud_id BIGINT NOT NULL,
ADD CONSTRAINT chk_solicitud_deportiva_edad CHECK (edad >= 0),
ADD CONSTRAINT chk_solicitud_deportiva_peso CHECK (peso >= 0);

ALTER TABLE invitacion
MODIFY club_id BIGINT NOT NULL,
MODIFY usuario_id BIGINT NOT NULL,
MODIFY rol VARCHAR(15) NOT NULL,
MODIFY estado VARCHAR(20) NOT NULL,
MODIFY created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP;

ALTER TABLE categoria
MODIFY club_id BIGINT NOT NULL,
MODIFY nombre VARCHAR(20) NOT NULL,
MODIFY created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP;

ALTER TABLE grupo_deportivo
MODIFY club_id BIGINT NOT NULL,
MODIFY nombre VARCHAR(25) NOT NULL,
MODIFY created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP;

ALTER TABLE entrenador
MODIFY usuario_club_id BIGINT NOT NULL,
MODIFY created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP;

ALTER TABLE acudiente
MODIFY nombre VARCHAR(20) NOT NULL,
MODIFY apellido VARCHAR(20) NOT NULL,
MODIFY created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP;

ALTER TABLE deportista
MODIFY usuario_club_id BIGINT NOT NULL,
MODIFY created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
ADD CONSTRAINT chk_deportista_peso CHECK (peso >= 0),
ADD CONSTRAINT chk_deportista_estatura CHECK (estatura >= 0);

ALTER TABLE entrenador_categoria
MODIFY entrenador_id BIGINT NOT NULL,
MODIFY categoria_id BIGINT NOT NULL;

ALTER TABLE grupo_entrenador
MODIFY grupo_id BIGINT NOT NULL,
MODIFY entrenador_id BIGINT NOT NULL;

ALTER TABLE sesion_entrenamiento
MODIFY grupo_id BIGINT NOT NULL,
MODIFY fecha DATE NOT NULL,
MODIFY created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP;

ALTER TABLE actividad
MODIFY club_id BIGINT NOT NULL,
MODIFY nombre VARCHAR(50) NOT NULL,
MODIFY created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP;

ALTER TABLE asistencia
MODIFY deportista_id BIGINT NOT NULL,
MODIFY sesion_id BIGINT NOT NULL,
MODIFY estado ENUM('presente','ausente') NOT NULL,
MODIFY fecha DATE NOT NULL,
MODIFY created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP;

ALTER TABLE metricas_sesion
MODIFY deportista_id BIGINT NOT NULL,
MODIFY entrenador_id BIGINT NOT NULL,
MODIFY sesion_id BIGINT NOT NULL,
MODIFY created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
ADD CONSTRAINT chk_metricas_tecnica CHECK (tecnica BETWEEN 1 AND 10),
ADD CONSTRAINT chk_metricas_rendimiento CHECK (rendimiento_fisico BETWEEN 1 AND 10),
ADD CONSTRAINT chk_metricas_tiempo CHECK (tiempo >= 0),
ADD CONSTRAINT chk_metricas_distancia CHECK (distancia >= 0),
ADD CONSTRAINT chk_metricas_velocidad CHECK (velocidad >= 0);
